    BackendStartup()
        BackendRun()
            PostgresMain()
                InitPostgres()
                exec_simple_query() 

exec_simple_query()
    PortalRun()
        PortalRunMulti()
            ProcessQuery()
                ExecutorRun()
                    standard_ExecutorRun()
                        ExecutePlan()
                            ExecProcNode()
